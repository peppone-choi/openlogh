object PositionCardRegistry {
    private val commandGroupMap: Map<String, CommandGroup> = buildMap {
        // 현재: 삼국지 커맨드 코드로 채워져 있음
        // "출병", "집합", ... → CommandGroup.OPERATIONS
    }

    fun canExecute(cards: Set<PositionCard>, actionCode: String): Boolean {
        val requiredGroup = commandGroupMap[actionCode] ?: return true // 매핑 없으면 허용
        return cards.any { card -> card.commandGroups.contains(requiredGroup) }
    }
}
