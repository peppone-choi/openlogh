Officer:

  factionId: Long             // 소속 진영
  planetId: Long              // 현재 위치 행성

<!-- PositionCard 추가/제거 패턴 -->
// Officer.positionCards는 쉼표 구분 문자열
fun addCard(officer: Officer, card: String) {
    val cards = officer.positionCards.split(",").filter { it.isNotBlank() }.toMutableSet()
    cards.add(card)
    officer.positionCards = cards.joinToString(",")
}
fun removeCard(officer: Officer, card: String) {
    val cards = officer.positionCards.split(",").filter { it.isNotBlank() }.toMutableSet()
    cards.remove(card)
    officer.positionCards = cards.joinToString(",")
}
