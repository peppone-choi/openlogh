       import com.openlogh.command.CommandCost
       import com.openlogh.command.CommandEnv
       import com.openlogh.command.CommandResult
       import com.openlogh.command.OfficerCommand

       /**
        * Phase 12 D-01/D-19: 작전계획 커맨드 — persists an OperationPlan entity.
        *
        * Args:
        *   - objective: String — "CONQUEST" | "DEFENSE" | "SWEEP"
        *   - targetStarSystemId: Long — the star system this operation targets (D-03)
        *   - participantFleetIds: List<Long> — fleets assigned to this operation
        *   - scale: Int? — MCP leverage, 1..7 (default 1)
        *   - planName: String? — display name (default "작전계획-{year}-{month}")
        *
        * Validation (D-19):
        *   - participantFleetIds non-empty
        *   - objective string is a valid MissionObjective enum
        *   - every fleet.factionId == general.factionId (ownership)
        *   - 1 fleet = 1 operation — atomically remove fleet from prior operation (D-04)
        */
       class OperationPlanCommand(
           general: Officer,
           env: CommandEnv,
           arg: Map<String, Any>?,
       ) : OfficerCommand(general, env, arg) {

           override val actionName: String = "작전계획"
           override fun getCost(): CommandCost = CommandCost()
           override fun getCommandPoolType(): StatCategory = StatCategory.MCP
           override fun getPreReqTurn(): Int = 0
           override fun getPostReqTurn(): Int = 0

           override suspend fun run(rng: Random): CommandResult {
               val nation = nation ?: return CommandResult.fail("소속 진영 정보를 찾을 수 없습니다.")
               val services = this.services ?: return CommandResult.fail("CommandServices unavailable")
               val opRepo = services.operationPlanRepository
                   ?: return CommandResult.fail("OperationPlanRepository unavailable")
               val fleetRepo = services.fleetRepository
                   ?: return CommandResult.fail("FleetRepository unavailable")

               // --- arg parsing ---
               val objectiveStr = arg?.get("objective") as? String
                   ?: return CommandResult.fail("objective 미지정")
               val objective = runCatching { MissionObjective.valueOf(objectiveStr) }
                   .getOrNull() ?: return CommandResult.fail("objective 값이 올바르지 않습니다: $objectiveStr")

               val targetStarSystemId = (arg["targetStarSystemId"] as? Number)?.toLong()
                   ?: return CommandResult.fail("targetStarSystemId 미지정")

               @Suppress("UNCHECKED_CAST")
               val fleetIdsRaw = arg["participantFleetIds"] as? List<*>
                   ?: return CommandResult.fail("participantFleetIds 미지정")
               val participantFleetIds = fleetIdsRaw.mapNotNull {
                   (it as? Number)?.toLong()
               }.toMutableList()
               if (participantFleetIds.isEmpty()) {
                   return CommandResult.fail("참가 부대를 1개 이상 지정해야 합니다")
               }

               val scale = (arg["scale"] as? Number)?.toInt() ?: 1
               val planName = arg["planName"] as? String
                   ?: "작전계획-${env.year}-${env.month}"

               // --- ownership validation ---
               val fleets = fleetRepo.findAllById(participantFleetIds)
               if (fleets.size != participantFleetIds.size) {
                   return CommandResult.fail("일부 부대를 찾을 수 없습니다")
               }
               val foreign = fleets.firstOrNull { it.factionId != general.factionId }
               if (foreign != null) {
                   return CommandResult.fail("부대 ${foreign.id}가 같은 진영이 아닙니다")
               }

               // --- 1-fleet-1-operation enforcement (D-04) ---
               // Read ALL active/pending ops for this faction, remove any prior
               // membership, save the updates, THEN persist the new operation —
               // all inside the surrounding @Transactional command executor.
               val priorOps = opRepo.findBySessionIdAndFactionIdAndStatusIn(
                   general.sessionId,
                   general.factionId,
                   listOf(OperationStatus.PENDING, OperationStatus.ACTIVE),
               )
               val participantSet = participantFleetIds.toSet()
               val now = OffsetDateTime.now()
               val mutatedPriors = mutableListOf<OperationPlan>()
