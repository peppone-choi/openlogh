    ```kotlin
    private fun validateEightStats(request: CreateGeneralRequest) {
        val stats = listOf(
            "leadership" to request.leadership.toInt(),
            "command" to (request.command?.toInt() ?: 50),
            "intelligence" to (request.intelligence?.toInt() ?: 50),
            "politics" to request.politics.toInt(),
            "administration" to (request.administration?.toInt() ?: 50),
            "mobility" to (request.mobility?.toInt() ?: 50),
            "attack" to (request.attack?.toInt() ?: 50),
            "defense" to (request.defense?.toInt() ?: 50),
        )
        val total = stats.sumOf { it.second }
        require(total == 400) { "8스탯 합계는 400이어야 합니다 (현재: $total)" }
        for ((name, value) in stats) {
            require(value in 20..95) { "$name 은 20~95 범위여야 합니다 (현재: $value)" }
        }
    }
    ```
