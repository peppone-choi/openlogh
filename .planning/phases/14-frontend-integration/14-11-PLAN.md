    import type { TacticalUnit, CommandHierarchyDto, BattleSide } from '@/types/tactical';
    import { findAlliesInMyChain } from './commandChain';

    export interface GhostEntry {
      x: number;
      y: number;
      tick: number;
      ships: number;
      unitType: string;
      side: BattleSide;
    }

    export const GHOST_TTL_TICKS = 60;
    export const GHOST_OPACITY_MAX = 0.4;
    export const GHOST_OPACITY_MIN = 0.15;
    export const GHOST_OPACITY_RAMP_START = 30;  // ticks

    /**
     * D-18 hierarchy-shared vision: compute which enemy fleetIds are currently visible
     * given the sensor cones of all allies in my command chain.
     */
    export function computeVisibleEnemies(
      units: TacticalUnit[],
      mySide: BattleSide,
      myOfficerId: number,
      hierarchy: CommandHierarchyDto | null | undefined
    ): Set<number> {
      const allies = findAlliesInMyChain(myOfficerId, hierarchy, units, mySide);
      const enemies = units.filter((u) => u.side !== mySide && u.isAlive);
      const visible = new Set<number>();
      for (const enemy of enemies) {
        for (const ally of allies) {
          const range = ally.sensorRange ?? 0;
          if (range <= 0) continue;
          const dx = enemy.posX - ally.posX;
          const dy = enemy.posY - ally.posY;
          const distSq = dx * dx + dy * dy;
          if (distSq <= range * range) {
            visible.add(enemy.fleetId);
            break;
          }
        }
      }
      return visible;
    }

    /**
     * D-20 fog update rules:
     * 1. For each currently-visible enemy → upsert lastSeen with current tick.
     * 2. For each ghost NOT in visible set → retain (stale preserved).
     * 3. For each dead enemy → remove (stop ghosting corpses).
     * 4. Prune entries older than GHOST_TTL_TICKS.
     */
    export function updateLastSeenEnemyPositions(
      prev: Record<number, GhostEntry>,
      units: TacticalUnit[],
      visibleEnemyIds: Set<number>,
      currentTick: number,
      mySide: BattleSide
    ): Record<number, GhostEntry> {
      const next: Record<number, GhostEntry> = { ...prev };

      // Upsert visible enemies
      for (const u of units) {
        if (u.side === mySide) continue;
        if (!u.isAlive) {
          delete next[u.fleetId];
          continue;
        }
        if (visibleEnemyIds.has(u.fleetId)) {
          next[u.fleetId] = {
            x: u.posX,
            y: u.posY,
            tick: currentTick,
            ships: u.ships,
            unitType: u.unitType,
            side: u.side,
          };
        }
      }

      // Prune old ghosts
      for (const fleetId of Object.keys(next)) {
        const id = Number(fleetId);
        const entry = next[id];
        if (currentTick - entry.tick > GHOST_TTL_TICKS) {
          delete next[id];
        }
      }

      return next;
    }

    /**
     * D-20 opacity ramp — 0.4 for fresh (0-30 ticks), ramps linearly to 0.15 by 60 ticks.
     */
    export function ghostOpacity(currentTick: number, entryTick: number): number {
      const age = currentTick - entryTick;
      if (age < 0) return GHOST_OPACITY_MAX;
      if (age <= GHOST_OPACITY_RAMP_START) return GHOST_OPACITY_MAX;
      if (age >= GHOST_TTL_TICKS) return GHOST_OPACITY_MIN;
      const progress = (age - GHOST_OPACITY_RAMP_START) / (GHOST_TTL_TICKS - GHOST_OPACITY_RAMP_START);
      return GHOST_OPACITY_MAX + (GHOST_OPACITY_MIN - GHOST_OPACITY_MAX) * progress;
    }
    ```

    import { computeVisibleEnemies, updateLastSeenEnemyPositions } from '@/lib/fogOfWar';

    onBattleTick: (data: BattleTickBroadcast) => {
      const myId = get().myOfficerId ?? -1;
      const myUnit = data.units.find((u) => u.officerId === myId);
      const mySide = myUnit?.side;

      set((state) => {
        const prevBattle = state.currentBattle;
        const attackerHierarchy = data.attackerHierarchy ?? prevBattle?.attackerHierarchy ?? null;
        const defenderHierarchy = data.defenderHierarchy ?? prevBattle?.defenderHierarchy ?? null;

        // Fog update — only if I'm in a side
        let lastSeen = state.lastSeenEnemyPositions;
        if (mySide) {
          const myHierarchy = mySide === 'ATTACKER' ? attackerHierarchy : defenderHierarchy;
          const visibleEnemies = computeVisibleEnemies(data.units, mySide, myId, myHierarchy);
          lastSeen = updateLastSeenEnemyPositions(
            state.lastSeenEnemyPositions,
            data.units,
            visibleEnemies,
            data.tickCount,
            mySide
          );
        }

        const nextBattle = prevBattle
          ? {
              ...prevBattle,
              tickCount: data.tickCount,
              phase: data.phase,
              result: data.result ?? prevBattle.result,
              units: data.units,
              attackerHierarchy,
              defenderHierarchy,
            }
          : null;

        return {
          units: data.units,
          recentEvents: [...data.events, ...state.recentEvents].slice(0, 50),
          currentBattle: nextBattle,
          myEnergy: myUnit?.energy ?? state.myEnergy,
          myFormation: myUnit?.formation ?? state.myFormation,
          lastSeenEnemyPositions: lastSeen,
          activeSuccessionFleetIds: state.activeSuccessionFleetIds,
          activeFlagshipDestroyedFleetIds: state.activeFlagshipDestroyedFleetIds.filter(
            (entry) => entry.expiresAt > Date.now()
          ),
        };
      });
    }

    'use client';

    import { useMemo } from 'react';
    import { useTacticalStore } from '@/stores/tacticalStore';
    import { ghostOpacity, computeVisibleEnemies, GHOST_OPACITY_RAMP_START } from '@/lib/fogOfWar';
    import { useOfficerStore } from '@/stores/officerStore';
    import { EnemyGhostIcon } from './EnemyGhostIcon';

    interface FogLayerProps {
      scaleX: number;
      scaleY: number;
    }

    /**
     * Phase 14 FE-05 — renders stale enemy ghosts inside BattleMap's fog-ghosts layer.
     * Currently-visible enemies are NOT rendered here (they render as live TacticalUnitIcon in units layer).
     */
    export function FogLayer({ scaleX, scaleY }: FogLayerProps) {
      const battle = useTacticalStore((s) => s.currentBattle);
      const lastSeen = useTacticalStore((s) => s.lastSeenEnemyPositions);
      const myOfficerId = useOfficerStore((s) => s.currentOfficerId ?? -1);

      const { currentTick, visibleIds } = useMemo(() => {
        if (!battle) return { currentTick: 0, visibleIds: new Set<number>() };
        const myUnit = battle.units.find((u) => u.officerId === myOfficerId);
        const mySide = myUnit?.side;
        if (!mySide) return { currentTick: battle.tickCount, visibleIds: new Set<number>() };
        const hierarchy = mySide === 'ATTACKER' ? battle.attackerHierarchy : battle.defenderHierarchy;
        return {
          currentTick: battle.tickCount,
          visibleIds: computeVisibleEnemies(battle.units, mySide, myOfficerId, hierarchy),
        };
      }, [battle, myOfficerId]);

      if (!battle) return null;

      return (
        <>
          {Object.entries(lastSeen).map(([idStr, entry]) => {
            const id = Number(idStr);
            if (visibleIds.has(id)) return null;       // currently visible → skip ghost
            const ticksAgo = currentTick - entry.tick;
            const opacity = ghostOpacity(currentTick, entry.tick);
            const isStale = ticksAgo > GHOST_OPACITY_RAMP_START;
            return (
              <EnemyGhostIcon
                key={id}
                fleetId={id}
                entry={entry}
                cx={entry.x * scaleX}
                cy={entry.y * scaleY}
                opacity={opacity}
                ticksAgo={ticksAgo}
                isStale={isStale}
              />
            );
          })}
        </>
      );
    }

    export default FogLayer;
    ```
