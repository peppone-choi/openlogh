    ```ts
    import type { TacticalUnit, CommandHierarchyDto } from '@/types/tactical';

    export type DragGateReason = 'WITHIN_CRC' | 'MOVING' | 'NO_HIERARCHY' | 'ALIVE_REQUIRED' | null;

    export interface DragGateResult {
      allowed: boolean;
      reason: DragGateReason;
      /** Korean tooltip copy for disabled chips */
      message?: string;
    }

    /**
     * D-06: Determine whether a unit may be re-assigned via drag in the current phase.
     *   - PREPARING: always allowed (if unit alive)
     *   - ACTIVE: CMD-05 rules — unit must be outside its current commander's CRC AND stopped
     */
    export function canReassignUnit(
      unit: TacticalUnit,
      phase: 'PREPARING' | 'ACTIVE' | 'ENDED',
      hierarchy: CommandHierarchyDto | null | undefined,
      commanderUnit: TacticalUnit | null   // unit's current sub-fleet commander, null if unassigned
    ): DragGateResult {
      if (!unit.isAlive) {
        return { allowed: false, reason: 'ALIVE_REQUIRED' };
      }
      if (phase === 'PREPARING') {
        return { allowed: true, reason: null };
      }
      if (phase === 'ENDED') {
        return { allowed: false, reason: 'WITHIN_CRC' };
      }
      if (!hierarchy) {
        return { allowed: false, reason: 'NO_HIERARCHY' };
      }
      // ACTIVE phase — CMD-05
      // Unassigned unit is always reassignable in ACTIVE
      if (!commanderUnit) {
        return { allowed: true, reason: null };
      }
      const dx = unit.posX - commanderUnit.posX;
      const dy = unit.posY - commanderUnit.posY;
      const distSq = dx * dx + dy * dy;
      const crcRadius = commanderUnit.commandRange ?? 0;
      const isOutsideCRC = distSq > crcRadius * crcRadius;
      if (!isOutsideCRC) {
        return {
          allowed: false,
          reason: 'WITHIN_CRC',
          message: '이 유닛은 교전 중이라 재배정할 수 없습니다.',
        };
      }
      // "Stopped" check — unit has no velocity. The frontend DTO does not expose velX/velY
      // (Phase 14 RESEARCH.md Open Question 2). For now, approximate via `isRetreating: false`
      // AND a server-provided `isStopped?: boolean` field. If absent, assume moving.
      // TODO: backend DTO extension may ship velX/velY; this helper will be updated.
      const isMoving = (unit as any).isStopped === false || ((unit as any).isStopped === undefined && !unit.isRetreating === false);
      // Conservative: if we cannot tell, treat as moving (safer — prevents mid-battle reassignment errors)
      const cannotTellStopped = (unit as any).isStopped === undefined;
      if (cannotTellStopped || isMoving) {
        return {
          allowed: false,
          reason: 'MOVING',
          message: '이 유닛은 교전 중이라 재배정할 수 없습니다.',
        };
      }
      return { allowed: true, reason: null };
    }
    ```
