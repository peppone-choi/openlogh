    ```
    useEffect(() => {
      if (!sessionId) return;
      const unsub = subscribeWebSocket('/topic/world/' + sessionId + '/operations', (msg) => {

        useGalaxyStore.getState().handleOperationEvent(evt);
      });
      return unsub;
    }, [sessionId]);
    ```
