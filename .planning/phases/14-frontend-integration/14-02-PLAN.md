    ```kotlin
    /**
     * Builds a merit breakdown for a completed battle.
     * @throws IllegalStateException if battle has not ended.
     * @throws NoSuchElementException if battle does not exist.
     */
    fun buildBattleSummary(sessionId: Long, battleId: Long): com.openlogh.dto.BattleSummaryDto {
        val battle = tacticalBattleRepository.findById(battleId)
            .orElseThrow { NoSuchElementException("Battle $battleId not found") }
        require(battle.sessionId == sessionId) { "Session mismatch" }
        require(battle.phase == com.openlogh.model.BattlePhase.ENDED) {
            "전투가 아직 종료되지 않았습니다"
        }

        val unitSnapshots = battle.unitSnapshots  // assumes TacticalBattle has persisted final unit state; if not, fetch from event log
        val operationParticipants = battle.operationParticipantFleetIds ?: emptyList()
        val winner = battle.result  // "ATTACKER_WIN" | "DEFENDER_WIN" | "DRAW" | null

        val rows = unitSnapshots.map { snap ->
            val officer = officerRepository.findById(snap.officerId).orElse(null)
            val onWinningSide = when (winner) {
                "ATTACKER_WIN" -> snap.side == "ATTACKER"
                "DEFENDER_WIN" -> snap.side == "DEFENDER"
                else -> false
            }
            val baseMerit = if (onWinningSide) {
                ((100.0 * snap.survivingShips / snap.maxShips).toInt()).coerceAtLeast(10)
            } else 0
            val isParticipant = operationParticipants.contains(snap.fleetId)
            val multiplier = if (isParticipant && baseMerit > 0) 1.5 else 1.0
            val totalMerit = (baseMerit * multiplier).toInt()

            com.openlogh.dto.BattleSummaryRow(
                fleetId = snap.fleetId,
                officerId = snap.officerId,
                officerName = officer?.name ?: "?",
                side = snap.side,
                survivingShips = snap.survivingShips,
                maxShips = snap.maxShips,
                baseMerit = baseMerit,
                operationMultiplier = multiplier,
                totalMerit = totalMerit,
                isOperationParticipant = isParticipant,
            )
        }

        return com.openlogh.dto.BattleSummaryDto(
            battleId = battle.id,
            winner = winner,
            durationTicks = battle.tickCount,
            rows = rows,
        )
    }
    ```
