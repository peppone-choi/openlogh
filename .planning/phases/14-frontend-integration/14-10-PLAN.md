    import type { CommandHierarchyDto, TacticalUnit, BattleSide } from '@/types/tactical';

    export interface VisibleCommander {
      officerId: number;
      officerName: string;
      isMine: boolean;
      isCommandable: boolean;
      side: BattleSide;
      /** Fleet this commander owns (their flagship fleetId) */
      flagshipFleetId: number;
    }

    /**
     * D-01: Determine which CRCs the logged-in officer should see on the tactical map.
     *   - Fleet commander → sees own CRC + every sub-fleet commander's CRC
     *   - Sub-fleet commander → sees only own CRC
     *   - Plain officer → sees nothing
     * @param myOfficerId logged-in officer id (-1 if not in a commander slot)
     * @param hierarchy   attackerHierarchy OR defenderHierarchy (whichever side I'm on)
     * @param units       current units (to map officerId → fleetId)
     * @param side        my battle side
     */
    export function findVisibleCrcCommanders(
      myOfficerId: number,
      hierarchy: CommandHierarchyDto | null | undefined,
      units: TacticalUnit[],
      side: BattleSide
    ): VisibleCommander[] {
      if (!hierarchy || myOfficerId < 0) return [];

      const findFleet = (officerId: number): TacticalUnit | undefined =>
        units.find((u) => u.officerId === officerId && u.side === side);

      const amFleetCommander = hierarchy.fleetCommander === myOfficerId
        || hierarchy.activeCommander === myOfficerId;
      const mySubFleet = hierarchy.subFleets.find((sf) => sf.commanderOfficerId === myOfficerId);

      const results: VisibleCommander[] = [];

      if (amFleetCommander) {
        // My own CRC
        const myUnit = findFleet(myOfficerId);
        if (myUnit) {
          results.push({
            officerId: myOfficerId,
            officerName: myUnit.officerName,
            isMine: true,
            isCommandable: true,
            side,
            flagshipFleetId: myUnit.fleetId,
          });
        }
        // All sub-commander CRCs (I can command them)
        for (const sf of hierarchy.subFleets) {
          const sfUnit = findFleet(sf.commanderOfficerId);
          if (sfUnit) {
            results.push({
              officerId: sf.commanderOfficerId,
              officerName: sf.commanderName,
              isMine: false,
              isCommandable: true,
              side,
              flagshipFleetId: sfUnit.fleetId,
            });
          }
        }
      } else if (mySubFleet) {
        const myUnit = findFleet(myOfficerId);
        if (myUnit) {
          results.push({
            officerId: myOfficerId,
            officerName: myUnit.officerName,
            isMine: true,
            isCommandable: true,
            side,
            flagshipFleetId: myUnit.fleetId,
          });
        }
      }

      return results;
    }

    /**
     * D-18: Hierarchy-shared vision — find all allied units within my command chain.
     * Used by fog-of-war to aggregate sensor coverage.
     */
    export function findAlliesInMyChain(
      myOfficerId: number,
      hierarchy: CommandHierarchyDto | null | undefined,
      units: TacticalUnit[],
      side: BattleSide
    ): TacticalUnit[] {
      if (!hierarchy || myOfficerId < 0) return [];
      const allySideUnits = units.filter((u) => u.side === side && u.isAlive);

      // Fleet commander or active commander → all units in the hierarchy
      if (hierarchy.fleetCommander === myOfficerId || hierarchy.activeCommander === myOfficerId) {
        return allySideUnits;
      }
      // Sub-fleet commander → own sub-fleet memberFleetIds + own flagship
      const mySubFleet = hierarchy.subFleets.find((sf) => sf.commanderOfficerId === myOfficerId);
      if (mySubFleet) {
        const memberIds = new Set(mySubFleet.memberFleetIds);
        return allySideUnits.filter((u) =>
          memberIds.has(u.fleetId) || u.officerId === myOfficerId
        );
      }
      return [];
    }
