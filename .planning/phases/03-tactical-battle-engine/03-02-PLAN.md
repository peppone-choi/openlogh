    ### MissileWeaponSystem.kt (신규)

    ```kotlin
    package com.openlogh.engine.tactical

    import com.openlogh.model.TacticalWeaponEvent
    import com.openlogh.model.TacticalWeaponType
    import kotlin.math.abs
    import kotlin.math.sqrt
    import kotlin.random.Random

    class MissileWeaponSystem {

        /**
         * 미사일 공격 처리. missileCount 1 소모.
         * 사거리 내 균일 위력. 아군 피격 없음(direct targeting).
         */
        fun processMissileAttack(
            source: TacticalUnit,
            target: TacticalUnit,
            state: TacticalBattleState,
            rng: Random = Random,
        ): TacticalWeaponEvent? {
            if (source.missileCount <= 0) return null
            val dist = distance(source, target)
            val missileRange = TacticalWeaponType.MISSILE.baseRange * 100.0  // 800 units
            if (dist > missileRange) return null

            source.missileCount--

            // 히트 확률: 센서 기반 80~95%
            val hitChance = (0.80 + source.energy.sensorMultiplier() * 0.15).coerceAtMost(0.95)
            if (rng.nextDouble() >= hitChance) {
                state.tickEvents.add(BattleTickEvent("missile_miss", source.fleetId, target.fleetId, 0, "미사일 빗나감"))
                return null
            }

            val rawDmg = TacticalWeaponType.MISSILE.baseDamage
            val attackMod = source.formation.attackModifier * source.stance.attackModifier * (source.attack / 50.0)
            val defenseMod = target.formation.defenseModifier * target.stance.defenseModifier * target.energy.shieldAbsorption()
            val finalDmg = ((rawDmg * attackMod) * (1.0 - defenseMod)).toInt().coerceAtLeast(1)

            target.hp -= finalDmg
            val shipLoss = (finalDmg.toDouble() / target.maxHp.coerceAtLeast(1) * target.maxShips).toInt()
            target.ships = (target.ships - shipLoss).coerceAtLeast(0)

            state.tickEvents.add(BattleTickEvent("missile_hit", source.fleetId, target.fleetId, finalDmg,
                "미사일 ${source.officerName}→${target.officerName} ($finalDmg 피해)"))

            return TacticalWeaponEvent(TacticalWeaponType.MISSILE, source.fleetId, target.fleetId, finalDmg, TacticalWeaponType.MISSILE.supplyCostPerUse, state.tickCount)
        }

        /**
         * 전투정(스파르타니안) 발진 처리.
         * 낮은 직접피해 + 이동속도 30% 감소 60틱 부여.
         * FIGHTER vs FIGHTER: 피해 2배.
         */
        fun processFighterAttack(
            source: TacticalUnit,
            target: TacticalUnit,
            state: TacticalBattleState,
            rng: Random = Random,
        ): TacticalWeaponEvent? {
            val dist = distance(source, target)
            val fighterRange = TacticalWeaponType.FIGHTER.baseRange * 100.0  // 600 units
            if (dist > fighterRange) return null
            // 전투정 운용 함종(carrier/torpedo carrier)만 발진 가능
            if (!source.unitType.contains("CARRIER", ignoreCase = true)) return null

            val hitChance = (0.65 + source.energy.sensorMultiplier() * 0.2).coerceAtMost(0.90)
            if (rng.nextDouble() >= hitChance) return null

            val isFighterVsFighter = target.unitType.contains("CARRIER", ignoreCase = true)
            val rawDmg = TacticalWeaponType.FIGHTER.baseDamage *
                if (isFighterVsFighter) TacticalWeaponType.FIGHTER_INTERCEPT_DAMAGE_MULTIPLIER.toInt() else 1

            target.hp -= rawDmg
            // 속도 디버프 부여
            target.fighterSpeedDebuffTicks = TacticalWeaponType.FIGHTER_DEBUFF_DURATION_TICKS

            state.tickEvents.add(BattleTickEvent("fighter_attack", source.fleetId, target.fleetId, rawDmg,
                "전투정 발진 → 속도저하${if(isFighterVsFighter) " (인터셉트 2배)" else ""}"))

            return TacticalWeaponEvent(TacticalWeaponType.FIGHTER, source.fleetId, target.fleetId, rawDmg,
                TacticalWeaponType.FIGHTER.supplyCostPerUse, state.tickCount,
                isIntercept = isFighterVsFighter,
                speedReduction = TacticalWeaponType.FIGHTER_SPEED_REDUCTION)
        }

        private fun distance(a: TacticalUnit, b: TacticalUnit) =
            sqrt((a.posX - b.posX) * (a.posX - b.posX) + (a.posY - b.posY) * (a.posY - b.posY))
    }
    ```

    ```kotlin
    package com.openlogh.engine.tactical

    import com.openlogh.engine.war.DetectionEngine
    import com.openlogh.engine.war.DetectionTarget
    import com.openlogh.engine.war.DetectorUnit
    import com.openlogh.model.DetectionCapability
    import org.springframework.stereotype.Component

    /**
     * Wraps DetectionEngine for use within TacticalBattleState.
     * Builds detectionMatrix per faction per tick.
     */
    @Component
    class DetectionService(
        private val detectionEngine: DetectionEngine = DetectionEngine(),
    ) {
        /**
         * 전체 전투 색적 매트릭스 갱신.
         * TacticalBattleState.detectionMatrix에 결과를 덮어쓴다.
         */
        fun updateDetectionMatrix(state: TacticalBattleState) {
            val aliveUnits = state.units.filter { it.isAlive }
            state.detectionMatrix.clear()

            for (side in listOf(BattleSide.ATTACKER, BattleSide.DEFENDER)) {
                val friendlies = aliveUnits.filter { it.side == side }
                val enemies = aliveUnits.filter { it.side != side }
                if (friendlies.isEmpty() || enemies.isEmpty()) continue

                val detectors = friendlies.map { u ->
                    val stationaryBonus = if (!u.stance.canMove) 0.2 else 0.0
                    DetectorUnit(
                        fleetId = u.fleetId,
                        factionId = u.factionId,
                        posX = u.posX,
                        posY = u.posY,
                        capability = DetectionCapability(
                            baseRange = 200.0 * u.energy.sensorMultiplier() * (1.0 + stationaryBonus),
                            precision = u.intelligence / 100.0,
                        ),
                        energy = u.energy,
                    )
                }
                val targets = enemies.map { u ->
                    DetectionTarget(
                        fleetId = u.fleetId,
                        factionId = u.factionId,
                        posX = u.posX,
                        posY = u.posY,
                        evasion = DetectionCapability(
                            baseRange = 0.0,
                            precision = u.energy.sensorMultiplier() * 0.3,  // 고 SENSOR = 전자전 회피
                        ),
                        unitType = u.unitType,
                    )
                }

                val results = detectionEngine.performFactionDetection(detectors, targets)
                // confirmed detection: precision >= 0.5 or detectingUnitCount >= 2
                for (detection in results) {
                    if (detection.precision >= 0.5 || detection.detectingUnitCount >= 2) {
                        val detectingOfficerIds = friendlies.map { it.officerId }
                        for (officerId in detectingOfficerIds) {
                            state.detectionMatrix.getOrPut(officerId) { mutableSetOf() }
                                .add(detection.targetFleetId)
                        }
                    }
                }
            }
        }
    }
    ```
