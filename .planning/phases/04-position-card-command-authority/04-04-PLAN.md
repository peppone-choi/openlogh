2. **API client** - Add proposal endpoints to `frontend/src/lib/api.ts`:
   ```typescript
   // Proposal API
   proposals: {
     submit: (generalId: number, data: SubmitProposalRequest) =>
       axios.post(`/api/proposals/submit/${generalId}`, data),
     resolve: (generalId: number, proposalId: number, data: { approved: boolean; reason?: string }) =>
       axios.post(`/api/proposals/resolve/${generalId}/${proposalId}`, data),
     pending: (generalId: number) =>
       axios.get<Proposal[]>(`/api/proposals/pending/${generalId}`),
     my: (generalId: number) =>
       axios.get<Proposal[]>(`/api/proposals/my/${generalId}`),
     eligibleApprovers: (generalId: number, actionCode: string) =>
       axios.get<EligibleApprover[]>(`/api/proposals/eligible-approvers/${generalId}`, { params: { actionCode } }),
   }
   ```
