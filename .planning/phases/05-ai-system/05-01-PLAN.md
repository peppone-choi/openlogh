```kotlin
package com.openlogh.engine.ai

data class CommandCandidate(val commandName: String, val score: Double)

object UtilityScorer {
    // CommandGroup -> stat weights mapping (which stats drive each group's score)
    // OPERATIONS: attack, command, mobility
    // PERSONAL: leadership, administration
    // COMMANDER: command, leadership
    // LOGISTICS: administration, defense
    // PERSONNEL: leadership, politics
    // POLITICS: politics, administration
    // INTELLIGENCE: intelligence, mobility

    fun scoreGroups(officer: OfficerSnapshot, trait: PersonalityTrait): Map<String, Double>
    // Returns map of commandGroup name -> utility score
    // Score = sum of relevant stats * personality weight multipliers for those stats
    // e.g., OPERATIONS score = (officer.attack * weights.attack + officer.command * weights.command + officer.mobility * weights.mobility) / 3.0

    fun rankCandidates(
        availableCommands: List<String>,  // commands the officer has position card access to
        officer: OfficerSnapshot,
        trait: PersonalityTrait,
        commandGroupOf: (String) -> String,  // lookup function: commandName -> groupName
    ): List<CommandCandidate>
    // For each available command, compute its group score, return sorted descending
}
```

```kotlin
@Service
class AiCommandBridge(
    private val commandExecutor: CommandExecutor,
    private val commandRegistry: Gin7CommandRegistry,
    private val positionCardRegistry: PositionCardRegistry,
) {
    /**
     * Execute the best-scoring available gin7 command for this officer.
     * Falls back to "대기" (always-allowed) if no scoreable command is available or all fail.
     * Returns the command name that was attempted.
     */
    fun executeAiCommand(officer: Officer, world: SessionState, trait: PersonalityTrait): String
}
```

Implementation of `executeAiCommand`:
1. Get available commands: filter `commandRegistry.allCommandNames()` by checking `positionCardRegistry` against officer's `positionCards` list — only keep commands the officer's held cards authorize
2. Call `UtilityScorer.rankCandidates(availableCommands, officer, trait, groupLookup)`
3. Try top-3 candidates in order: call `commandExecutor.executeGeneralCommand(officer.id, commandName, emptyMap(), world)` — if `CommandResult.success` is true, return commandName
4. Fallback: execute "대기", return "대기"
5. Catch all exceptions, log warn, return "대기"
