    ```kotlin
    @Service
    class CharacterLifecycleService(
        private val officerRepository: OfficerRepository,
    ) {
        companion object {
            const val MAX_DELETION_RANK = 4  // colonel
            const val MAX_INHERITANCE_AGE = 60
            const val INJURY_STAT_DIVISOR = 10  // injury value / 10 = stat penalty
        }

        /**
         * CHAR-10: Check if officer can be deleted.
         * Conditions: rank <= colonel (4), currently on a planet (residence/hotel).
         * The specific facility check (residence vs hotel) is deferred to location system.
         */
        fun canDeleteOfficer(officer: Officer): Boolean {
            if (officer.rank > MAX_DELETION_RANK) return false
            if (officer.locationState != "planet") return false
            if (officer.killTurn != null) return false  // already dead
            return true
        }

        /**
         * CHAR-10: Execute officer deletion.
         * Sets killTurn to mark as dead, clears userId to release the slot.
         */
        @Transactional
        fun deleteOfficer(officer: Officer, currentTurn: Int): Officer {
            require(canDeleteOfficer(officer)) { "삭제 조건을 충족하지 않습니다." }
            officer.killTurn = currentTurn
            officer.userId = null
            return officerRepository.save(officer)
        }

        /**
         * CHAR-11: Apply injury after battle defeat.
         * injury value determines stat penalty: each stat reduced by injury / INJURY_STAT_DIVISOR.
         * Stats cannot go below 1. Pre-injury stats stored in meta for recovery.
         */
        @Transactional
        fun applyInjury(officer: Officer, injuryLevel: Int): Officer {
            require(injuryLevel > 0) { "부상 레벨은 1 이상이어야 합니다." }

            // Store pre-injury stats in meta for recovery
            val preInjury = mapOf(
                "preInjuryLeadership" to officer.leadership,
                "preInjuryCommand" to officer.command,
                "preInjuryIntelligence" to officer.intelligence,
                "preInjuryPolitics" to officer.politics,
                "preInjuryAdministration" to officer.administration,
                "preInjuryMobility" to officer.mobility,
                "preInjuryAttack" to officer.attack,
                "preInjuryDefense" to officer.defense,
            )
            officer.meta.putAll(preInjury)

            officer.injury = injuryLevel
            val penalty = injuryLevel / INJURY_STAT_DIVISOR
            officer.leadership = (officer.leadership - penalty).coerceAtLeast(1)
            officer.command = (officer.command - penalty).coerceAtLeast(1)
            officer.intelligence = (officer.intelligence - penalty).coerceAtLeast(1)
            officer.politics = (officer.politics - penalty).coerceAtLeast(1)
            officer.administration = (officer.administration - penalty).coerceAtLeast(1)
            officer.mobility = (officer.mobility - penalty).coerceAtLeast(1)
            officer.attack = (officer.attack - penalty).coerceAtLeast(1)
            officer.defense = (officer.defense - penalty).coerceAtLeast(1)

            return officerRepository.save(officer)
        }

        /**
         * CHAR-11: Recover from injury. Restores pre-injury stats from meta.
         */
        @Transactional
        fun recoverFromInjury(officer: Officer): Officer {
            if (officer.injury == 0) return officer

            // Restore from meta
            officer.leadership = (officer.meta["preInjuryLeadership"] as? Int) ?: officer.leadership
            officer.command = (officer.meta["preInjuryCommand"] as? Int) ?: officer.command
            officer.intelligence = (officer.meta["preInjuryIntelligence"] as? Int) ?: officer.intelligence
            officer.politics = (officer.meta["preInjuryPolitics"] as? Int) ?: officer.politics
            officer.administration = (officer.meta["preInjuryAdministration"] as? Int) ?: officer.administration
            officer.mobility = (officer.meta["preInjuryMobility"] as? Int) ?: officer.mobility
            officer.attack = (officer.meta["preInjuryAttack"] as? Int) ?: officer.attack
            officer.defense = (officer.meta["preInjuryDefense"] as? Int) ?: officer.defense

            // Clear injury state
            officer.injury = 0
            officer.meta.keys.removeAll { it.startsWith("preInjury") }

            return officerRepository.save(officer)
        }

        /**
         * CHAR-12: Trigger officer death on flagship destruction.
         * Sets killTurn, moves officer to home planet (or faction capital if no home set).
         */
        @Transactional
        fun triggerDeath(officer: Officer, currentTurn: Int, factionCapitalPlanetId: Long): Officer {
            officer.killTurn = currentTurn
            val returnPlanetId = officer.homePlanetId ?: factionCapitalPlanetId
            officer.planetId = returnPlanetId
            officer.locationState = "planet"
            return officerRepository.save(officer)
        }

        /**
         * CHAR-09: Check if officer can be inherited to a new session.
         * Conditions: has fame/evaluation points, age <= 60, not dead.
         */
        fun canInherit(officer: Officer): Boolean {
            if (officer.killTurn != null) return false
            if (officer.age > MAX_INHERITANCE_AGE) return false
            if (officer.famePoints <= 0) return false
            return true
        }

        /**
         * CHAR-09: Create an inherited officer in a new session.
         * Copies identity and fame; stats carry over at 80% of original.
         */
        @Transactional
        fun inheritOfficer(
            sourceOfficer: Officer, targetSessionId: Long, userId: Long,
            planetId: Long, starSystemId: Long,
        ): Officer {
            require(canInherit(sourceOfficer)) { "인계 조건을 충족하지 않습니다." }
            val inherited = Officer().apply {
                this.worldId = targetSessionId
                this.userId = userId
                this.factionId = sourceOfficer.factionId
                this.name = sourceOfficer.name
                this.picture = sourceOfficer.picture
                // Stats at 80% of original
                this.leadership = (sourceOfficer.leadership * 0.8).toInt()
                this.command = (sourceOfficer.command * 0.8).toInt()
                this.intelligence = (sourceOfficer.intelligence * 0.8).toInt()
                this.politics = (sourceOfficer.politics * 0.8).toInt()
                this.administration = (sourceOfficer.administration * 0.8).toInt()
                this.mobility = (sourceOfficer.mobility * 0.8).toInt()
                this.attack = (sourceOfficer.attack * 0.8).toInt()
                this.defense = (sourceOfficer.defense * 0.8).toInt()
                this.originType = sourceOfficer.originType
                this.careerType = sourceOfficer.careerType
                this.rank = 0  // start at sub-lieutenant in new session
                this.planetId = planetId
                this.starSystemId = starSystemId
                this.locationState = "planet"
                this.homePlanetId = planetId
                this.famePoints = sourceOfficer.famePoints
                this.npcState = 0
            }
            return officerRepository.save(inherited)
        }
    }
    ```
