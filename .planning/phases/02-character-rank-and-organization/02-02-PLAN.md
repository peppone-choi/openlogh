    ```kotlin
    @Service
    class CharacterCreationService(
        private val officerRepository: OfficerRepository,
    ) {
        companion object {
            const val STAT_TOTAL = 400
            const val STAT_MIN = 10
            const val STAT_MAX = 100
            val STAT_KEYS = listOf("leadership", "command", "intelligence", "politics",
                "administration", "mobility", "attack", "defense")
            val EMPIRE_ORIGINS = setOf("noble", "knight", "commoner")
            val ALLIANCE_ORIGINS = setOf("citizen")
        }

        data class StatAllocation(
            val leadership: Int, val command: Int, val intelligence: Int,
            val politics: Int, val administration: Int,
            val mobility: Int, val attack: Int, val defense: Int,
        )

        fun validateStatAllocation(stats: StatAllocation): Boolean {
            val values = listOf(stats.leadership, stats.command, stats.intelligence,
                stats.politics, stats.administration, stats.mobility, stats.attack, stats.defense)
            if (values.any { it < STAT_MIN || it > STAT_MAX }) return false
            if (values.sum() != STAT_TOTAL) return false
            return true
        }

        fun validateOrigin(factionType: String, originType: String): Boolean {
            return when (factionType) {
                "empire" -> originType in EMPIRE_ORIGINS
                "alliance" -> originType in ALLIANCE_ORIGINS
                else -> false
            }
        }

        @Transactional
        fun createGeneratedOfficer(
            sessionId: Long, userId: Long, factionId: Long,
            name: String, stats: StatAllocation, originType: String,
            factionType: String, planetId: Long, starSystemId: Long,
        ): Officer {
            require(validateStatAllocation(stats)) { "능력치 합계는 ${STAT_TOTAL}이어야 합니다." }
            require(validateOrigin(factionType, originType)) { "유효하지 않은 출자입니다." }
            require(name.length in 2..20) { "이름은 2~20자여야 합니다." }

            val officer = Officer().apply {
                this.worldId = sessionId
                this.userId = userId
                this.factionId = factionId
                this.name = name
                this.leadership = stats.leadership
                this.command = stats.command
                this.intelligence = stats.intelligence
                this.politics = stats.politics
                this.administration = stats.administration
                this.mobility = stats.mobility
                this.attack = stats.attack
                this.defense = stats.defense
                this.originType = originType
                this.careerType = "military"
                this.rank = 0  // sub-lieutenant
                this.planetId = planetId
                this.starSystemId = starSystemId
                this.locationState = "planet"
                this.homePlanetId = planetId
                this.npcState = 0  // player-controlled
            }
            return officerRepository.save(officer)
        }

        @Transactional
        fun selectOriginalOfficer(
            sessionId: Long, userId: Long, officerId: Long,
        ): Officer {
            val officer = officerRepository.findById(officerId)
                .orElseThrow { IllegalArgumentException("캐릭터를 찾을 수 없습니다.") }
            require(officer.worldId == sessionId) { "다른 세션의 캐릭터입니다." }
            require(officer.userId == null || officer.userId == 0L) {
                "이미 다른 플레이어가 선택한 캐릭터입니다."
            }
            officer.userId = userId
            officer.npcState = 0
            return officerRepository.save(officer)
        }

        fun getAvailableOriginals(sessionId: Long, factionId: Long): List<Officer> {
            return officerRepository.findByWorldIdAndFactionId(sessionId, factionId)
                .filter { (it.userId == null || it.userId == 0L) && it.npcState != 0 }
        }
    }
    ```

    1. Create `CharacterController.kt`:
    ```kotlin
    @RestController
    @RequestMapping("/api/character")
    class CharacterController(
        private val characterCreationService: CharacterCreationService,
    ) {
        data class GenerateRequest(
            val sessionId: Long,
            val factionId: Long,
            val factionType: String,
            val name: String,
            val originType: String,
            val stats: CharacterCreationService.StatAllocation,
            val planetId: Long,
            val starSystemId: Long,
        )

        data class SelectOriginalRequest(
            val sessionId: Long,
            val officerId: Long,
        )

        @PostMapping("/generate")
        fun generateCharacter(
            @RequestBody request: GenerateRequest,
            @AuthenticationPrincipal userId: Long,
        ): ResponseEntity<Officer> {
            val officer = characterCreationService.createGeneratedOfficer(
                sessionId = request.sessionId,
                userId = userId,
                factionId = request.factionId,
                name = request.name,
                stats = request.stats,
                originType = request.originType,
                factionType = request.factionType,
                planetId = request.planetId,
                starSystemId = request.starSystemId,
            )
            return ResponseEntity.status(201).body(officer)
        }

        @PostMapping("/select-original")
        fun selectOriginal(
            @RequestBody request: SelectOriginalRequest,
            @AuthenticationPrincipal userId: Long,
        ): ResponseEntity<Officer> {
            val officer = characterCreationService.selectOriginalOfficer(
                sessionId = request.sessionId,
                userId = userId,
                officerId = request.officerId,
            )
            return ResponseEntity.ok(officer)
        }

        @GetMapping("/available-originals")
        fun getAvailableOriginals(
            @RequestParam sessionId: Long,
            @RequestParam factionId: Long,
        ): ResponseEntity<List<Officer>> {
            val officers = characterCreationService.getAvailableOriginals(sessionId, factionId)
            return ResponseEntity.ok(officers)
        }
    }
    ```
