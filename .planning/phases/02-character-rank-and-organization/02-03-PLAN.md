    ```kotlin
    @Service
    class StatGrowthService {

        companion object {
            const val EXP_THRESHOLD = 100
            const val STAT_CAP = 100
            const val YOUTH_AGE = 30
            const val ELDER_AGE = 50
            const val DECAY_AGE = 55
            const val YOUTH_MULTIPLIER = 1.2
            const val ELDER_MULTIPLIER = 0.8
            const val DECAY_CHANCE_PER_STAT = 0.01  // 1% per stat per tick
        }

        /**
         * Returns the growth rate multiplier based on officer age (CHAR-04).
         * Youth (< 30): +20% bonus
         * Prime (30-50): normal
         * Elder (> 50): -20% penalty
         */
        fun getAgeMultiplier(age: Int): Double {
            return when {
                age < YOUTH_AGE -> YOUTH_MULTIPLIER
                age > ELDER_AGE -> ELDER_MULTIPLIER
                else -> 1.0
            }
        }

        /**
         * Applies age-modified exp to an officer's stat exp field.
         * Called when CP is consumed (wiring in Phase 3).
         * rawExp is the base exp gained; age modifier is applied here.
         */
        fun addExp(officer: Officer, statName: String, rawExp: Int) {
            val multiplier = getAgeMultiplier(officer.age)
            val actualExp = (rawExp * multiplier).toInt().coerceAtLeast(1)
            when (statName) {
                "leadership" -> officer.leadershipExp += actualExp
                "command" -> officer.commandExp += actualExp
                "intelligence" -> officer.intelligenceExp += actualExp
                "politics" -> officer.politicsExp += actualExp
                "administration" -> officer.administrationExp += actualExp
                "mobility" -> officer.mobilityExp += actualExp
                "attack" -> officer.attackExp += actualExp
                "defense" -> officer.defenseExp += actualExp
            }
        }

        /**
         * Processes all 8 stat exp fields (CHAR-05).
         * When any *Exp field >= 100, increase corresponding stat by 1
         * and subtract 100 from exp (carrying over excess).
         * Stat cannot exceed STAT_CAP (100).
         * Returns list of stats that grew.
         */
        fun processExpGrowth(officer: Officer): List<String> {
            val grew = mutableListOf<String>()

            if (officer.leadershipExp >= EXP_THRESHOLD && officer.leadership < STAT_CAP) {
                officer.leadership++; officer.leadershipExp -= EXP_THRESHOLD; grew.add("leadership")
            }
            if (officer.commandExp >= EXP_THRESHOLD && officer.command < STAT_CAP) {
                officer.command++; officer.commandExp -= EXP_THRESHOLD; grew.add("command")
            }
            if (officer.intelligenceExp >= EXP_THRESHOLD && officer.intelligence < STAT_CAP) {
                officer.intelligence++; officer.intelligenceExp -= EXP_THRESHOLD; grew.add("intelligence")
            }
            if (officer.politicsExp >= EXP_THRESHOLD && officer.politics < STAT_CAP) {
                officer.politics++; officer.politicsExp -= EXP_THRESHOLD; grew.add("politics")
            }
            if (officer.administrationExp >= EXP_THRESHOLD && officer.administration < STAT_CAP) {
                officer.administration++; officer.administrationExp -= EXP_THRESHOLD; grew.add("administration")
            }
            if (officer.mobilityExp >= EXP_THRESHOLD && officer.mobility < STAT_CAP) {
                officer.mobility++; officer.mobilityExp -= EXP_THRESHOLD; grew.add("mobility")
            }
            if (officer.attackExp >= EXP_THRESHOLD && officer.attack < STAT_CAP) {
                officer.attack++; officer.attackExp -= EXP_THRESHOLD; grew.add("attack")
            }
            if (officer.defenseExp >= EXP_THRESHOLD && officer.defense < STAT_CAP) {
                officer.defense++; officer.defenseExp -= EXP_THRESHOLD; grew.add("defense")
            }

            return grew
        }

        /**
         * Elder stat decay (CHAR-04 extension).
         * Officers age > 55 have a 1% chance per stat per tick to lose 1 point.
         * Returns list of stats that decayed.
         */
        fun processElderDecay(officer: Officer, random: java.util.Random = java.util.Random()): List<String> {
            if (officer.age <= DECAY_AGE) return emptyList()
            val decayed = mutableListOf<String>()
            val stats = listOf(
                "leadership" to officer.leadership,
                "command" to officer.command,
                "intelligence" to officer.intelligence,
                "politics" to officer.politics,
                "administration" to officer.administration,
                "mobility" to officer.mobility,
                "attack" to officer.attack,
                "defense" to officer.defense,
            )
            for ((name, value) in stats) {
                if (value > 1 && random.nextDouble() < DECAY_CHANCE_PER_STAT) {
                    when (name) {
                        "leadership" -> officer.leadership--
                        "command" -> officer.command--
                        "intelligence" -> officer.intelligence--
                        "politics" -> officer.politics--
                        "administration" -> officer.administration--
                        "mobility" -> officer.mobility--
                        "attack" -> officer.attack--
                        "defense" -> officer.defense--
                    }
                    decayed.add(name)
                }
            }
            return decayed
        }
    }
    ```
